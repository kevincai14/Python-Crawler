    function toGate(blet){
    	var gate="";
    	if(blet!=''){
    		var blets =blet.split(",");
    		if(blets.length>0){
    			for(var i=0;i<blets.length;i++){
    				if(blets[i]==1 || blets[i]==2 || blets[i]==3){
    					gate=gate+"A"+",";
    				}if(blets[i]==21 || blets[i]==22 || blets[i]==23){
    					gate=gate+"G"+",";
    				}if(blets[i]==4 || blets[i]==5 ){
    					gate=gate+"B"+",";
    				}if(blets[i]==6 || blets[i]==7 ){
    					gate=gate+"C"+",";
    				}if(blets[i]==8 || blets[i]==9 ){
    					gate=gate+"D"+",";
    				}    				
    			}
    		}
    		if(gate.split(",").length>1){
    			gate=gate.substring(0,gate.length-1);
    		}
    	}
    	return gate;
    }
